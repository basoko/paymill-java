001    package com.paymill.services;
002    
003    import java.util.List;
004    
005    import javax.ws.rs.core.MultivaluedMap;
006    
007    import com.paymill.models.Payment;
008    import com.paymill.models.PaymillList;
009    import com.paymill.models.Preauthorization;
010    import com.paymill.models.Transaction;
011    import com.sun.jersey.core.util.MultivaluedMapImpl;
012    
013    /**
014     * The {@link PreauthorizationService} is used to list, create and delete PAYMILL {@link Preauthorization}s.
015     * @author Vassil Nikolov
016     * @since 3.0.0
017     */
018    public class PreauthorizationService extends AbstractService {
019    
020      private final static String PATH = "/preauthorizations";
021    
022      private PreauthorizationService( com.sun.jersey.api.client.Client httpClient ) {
023        super( httpClient );
024      }
025    
026      /**
027       * This function returns a {@link List} of PAYMILL {@link Preauthorization} objects.
028       * @return {@link PaymillList} which contains a {@link List} of PAYMILL {@link Preauthorization}s and their total count.
029       */
030      public PaymillList<Preauthorization> list() {
031        return this.list( null, null, null, null );
032      }
033    
034      /**
035       * This function returns a {@link List} of PAYMILL {@link Preauthorization} objects, overriding the default count and offset.
036       * @param count
037       *          Max {@link Integer} of returned objects in the {@link PaymillList}
038       * @param offset
039       *          {@link Integer} to start from.
040       * @return {@link PaymillList} which contains a {@link List} of PAYMILL {@link Preauthorization}s and their total count.
041       */
042      public PaymillList<Preauthorization> list( Integer count, Integer offset ) {
043        return this.list( null, null, count, offset );
044      }
045    
046      /**
047       * This function returns a {@link List} of PAYMILL {@link Preauthorization} objects. In which order this list is returned
048       * depends on the optional parameters. If <code>null</code> is given, no filter or order will be applied.
049       * @param filter
050       *          {@link Preauthorization.Filter} or <code>null</code>
051       * @param order
052       *          {@link Preauthorization.Order} or <code>null</code>
053       * @return {@link PaymillList} which contains a {@link List} of PAYMILL {@link Preauthorization}s and their total count.
054       */
055      public PaymillList<Preauthorization> list( Preauthorization.Filter filter, Preauthorization.Order order ) {
056        return this.list( filter, order, null, null );
057      }
058    
059      /**
060       * This function returns a {@link List} of PAYMILL {@link Preauthorization} objects. In which order this list is returned
061       * depends on the optional parameters. If <code>null</code> is given, no filter or order will be applied, overriding the default
062       * count and offset.
063       * @param filter
064       *          {@link Preauthorization.Filter} or <code>null</code>
065       * @param order
066       *          {@link Preauthorization.Order} or <code>null</code>
067       * @param count
068       *          Max {@link Integer} of returned objects in the {@link PaymillList}
069       * @param offset
070       *          {@link Integer} to start from.
071       * @return {@link PaymillList} which contains a {@link List} of PAYMILL {@link Preauthorization}s and their total count.
072       */
073      public PaymillList<Preauthorization> list( Preauthorization.Filter filter, Preauthorization.Order order, Integer count, Integer offset ) {
074        return RestfulUtils.list( PreauthorizationService.PATH, filter, order, count, offset, Preauthorization.class, super.httpClient );
075      }
076    
077      /**
078       * Returns and refresh data of a specific {@link Preauthorization}.
079       * @param preauthorization
080       *          A {@link Preauthorization} with Id.
081       * @return Refreshed instance of the given {@link Preauthorization}.
082       */
083      public Preauthorization get( Preauthorization preauthorization ) {
084        return RestfulUtils.show( PreauthorizationService.PATH, preauthorization, Preauthorization.class, super.httpClient );
085      }
086    
087      /**
088       * Returns and refresh data of a specific {@link Preauthorization}.
089       * @param preauthorizationId
090       *          Id of the {@link Preauthorization}.
091       * @return {@link Preauthorization} object, which represents a PAYMILL preauthorization.
092       */
093      public Preauthorization get( String preauthorizationId ) {
094        return this.get( new Preauthorization( preauthorizationId ) );
095      }
096    
097      /**
098       * Creates Use either a token or an existing payment to Authorizes the given amount with the given token.
099       * @param token
100       *          The identifier of a token.
101       * @param amount
102       *          Amount (in cents) which will be charged.
103       * @param currency
104       *          ISO 4217 formatted currency code.
105       * @return {@link Transaction} object with the {@link Preauthorization} as sub object.
106       */
107      public Transaction createWithToken( String token, Integer amount, String currency ) {
108        ValidationUtils.validatesToken( token );
109        ValidationUtils.validatesAmount( amount );
110        ValidationUtils.validatesCurrency( currency );
111    
112        MultivaluedMap<String, String> params = new MultivaluedMapImpl();
113    
114        params.add( "token", token );
115        params.add( "amount", String.valueOf( amount ) );
116        params.add( "currency", currency );
117    
118        return RestfulUtils.create( PreauthorizationService.PATH, params, Transaction.class, super.httpClient );
119      }
120    
121      /**
122       * Authorizes the given amount with the given {@link Payment}.<br />
123       * <strong>Works only for credit cards. Direct debit not supported.</strong>
124       * @param payment
125       *          The {@link Payment} itself (only creditcard-object)
126       * @param amount
127       *          Amount (in cents) which will be charged.
128       * @param currency
129       *          ISO 4217 formatted currency code.
130       * @return {@link Transaction} object with the {@link Preauthorization} as sub object.
131       */
132      public Transaction createWithPayment( Payment payment, Integer amount, String currency ) {
133        ValidationUtils.validatesPayment( payment );
134        ValidationUtils.validatesAmount( amount );
135        ValidationUtils.validatesCurrency( currency );
136    
137        MultivaluedMap<String, String> params = new MultivaluedMapImpl();
138    
139        params.add( "payment", payment.getId() );
140        params.add( "amount", String.valueOf( amount ) );
141        params.add( "currency", currency );
142    
143        return RestfulUtils.create( PreauthorizationService.PATH, params, Transaction.class, super.httpClient );
144      }
145    
146      /**
147       * This function deletes a preauthorization.
148       * @param preauthorization
149       *          The {@link Preauthorization} object to be deleted.
150       */
151      public void delete( Preauthorization preauthorization ) {
152        RestfulUtils.delete( PreauthorizationService.PATH, preauthorization, Preauthorization.class, super.httpClient );
153      }
154    
155      /**
156       * This function deletes a preauthorization.
157       * @param preauthorizationId
158       *          The Id of the {@link Preauthorization}.
159       */
160      public void delete( String preauthorizationId ) {
161        this.delete( new Preauthorization( preauthorizationId ) );
162      }
163    
164      /**
165       * This function deletes a preauthorization.
166       * @param transaction
167       *          A {@link Transaction} which should have a {@link Preauthorization} as a sub object.
168       */
169      public void delete( Transaction transaction ) {
170        this.delete( transaction.getPreauthorization() );
171      }
172    
173    }
