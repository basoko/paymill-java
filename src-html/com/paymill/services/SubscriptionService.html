100      /**
101       * This function creates a {@link Subscription} between a {@link Client} and an {@link Offer}. A {@link Client} can have several
102       * {@link Subscription}s to different {@link Offer}s, but only one {@link Subscription} to the same {@link Offer}. The
103       * {@link Client}s is charged for each billing interval entered. <br />
104       * <strong>NOTE</strong><br />
105       * This call will use the {@link Client} from the {@link Payment} object.
106       * @param offer
107       *          An {@link Offer} to subscribe to.
108       * @param payment
109       *          A {@link Payment} used for charging.
110       * @return {@link Subscription}, which allows you to charge recurring payments.
111       */
112      public Subscription createWithOfferAndPayment( Offer offer, Payment payment ) {
113        ValidationUtils.validatesOffer( offer );
114        ValidationUtils.validatesPayment( payment );
115    
116        MultivaluedMap<String, String> params = new MultivaluedMapImpl();
117        params.add( "offer", offer.getId() );
118        params.add( "payment", payment.getId() );
119    
120        return RestfulUtils.create( SubscriptionService.PATH, params, Subscription.class, super.httpClient );
121      }

195      /**
196       * This function creates a {@link Subscription} between a {@link Client} and an {@link Offer}. A {@link Client} can have several
197       * {@link Subscription}s to different {@link Offer}s, but only one {@link Subscription} to the same {@link Offer}. The
198       * {@link Client}s is charged for each billing interval entered. <br />
199       * <strong>NOTE</strong><br />
200       * If {@link Client} not provided the {@link Client} from the payment is being used.
201       * @param offer
202       *          An {@link Offer} to subscribe to.
203       * @param payment
204       *          A {@link Payment} used for charging.
205       * @param client
206       *          A {@link Client} to subscribe.
207       * @param trialStart
208       *          Date representing trial period start.
209       * @return {@link Subscription}, which allows you to charge recurring payments.
210       */
211      public Subscription createWithOfferPaymentAndClient( Offer offer, Payment payment, Client client, Date trialStart ) {
212        ValidationUtils.validatesOffer( offer );
213        ValidationUtils.validatesPayment( payment );
214        ValidationUtils.validatesClient( client );
215    
216        MultivaluedMap<String, String> params = new MultivaluedMapImpl();
217        params.add( "offer", offer.getId() );
218        params.add( "payment", payment.getId() );
219        params.add( "client", client.getId() );
220        if( trialStart != null ) {
221          params.add( "start_at", String.valueOf( trialStart.getTime() ) );
222        }
223    
224        return RestfulUtils.create( SubscriptionService.PATH, params, Subscription.class, super.httpClient );
225      }
