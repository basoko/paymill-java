011    /**
012     * If you’d like to reserve some money from the client’s credit card but you’d also like to execute the transaction itself a bit
013     * later, then use preauthorizations. This is NOT possible with direct debit. <br />
014     * <strong>A preauthorization is valid for 7 days.</strong>

016     * @since 3.0.0
017     */

189      public enum Status {
190        PREAUTH("preauth"), OPEN("open"), CLOSED("closed"), DELETED("deleted"), FAILED("failed"), PENDING("pending");
191    
192        private String value;
193    
194        private Status( final String value ) {
195          this.value = value;
196        }
197    
198        @JsonValue
199        public String getValue() {
200          return this.value;
201        }
202    
203        @JsonCreator
204        public static Status create( final String value ) {
205          for( Status status : Status.values() ) {
206            if( status.getValue().equals( value ) ) {
207              return status;
208            }
209          }
210          throw new IllegalArgumentException( "Invalid value for Preauthorization.Status" );
211        }
212      }

222      public final static class Filter {
223    
224        @SnakeCase( "client" )
225        private String clientId;
226    
227        @SnakeCase( "payment" )
228        private String paymentId;
229    
230        @SnakeCase( "amount" )
231        private String amount;
232    
233        @SnakeCase( "created_at" )
234        private String createdAt;
235    
236        private Filter() {
237          super();
238        }
239    
240        public Preauthorization.Filter byClientId( final String clientId ) {
241          this.clientId = clientId;
242          return this;
243        }
244    
245        public Preauthorization.Filter byPaymentId( final String paymentId ) {
246          this.paymentId = paymentId;
247          return this;
248        }
249    
250        public Preauthorization.Filter byAmount( final int amount ) {
251          this.amount = String.valueOf( amount );
252          return this;
253        }
254    
255        public Preauthorization.Filter byAmountGreaterThan( final int amount ) {
256          this.amount = ">" + String.valueOf( amount );
257          return this;
258        }
259    
260        public Preauthorization.Filter byAmountLessThan( final int amount ) {
261          this.amount = "<" + String.valueOf( amount );
262          return this;
263        }
264    
265        public Preauthorization.Filter byCreatedAt( final Date startCreatedAt, final Date endCreatedAt ) {
266          this.createdAt = String.valueOf( startCreatedAt.getTime() ) + "-" + String.valueOf( endCreatedAt.getTime() );
267          return this;
268        }
269      }
