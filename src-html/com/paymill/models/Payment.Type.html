352      public enum CardType {
353        VISA("visa"), MASTERCARD("mastercard"), MASTRO("maestro"), AMEX("amex"), JCB("jcb"),
354    
355        DINERS("diners"), DISCOVER("discover"), CHINA_UNION_PAY("china_union_pay"), UNKNOWN("unknown");
356    
357        private String value;
358    
359        private CardType( final String value ) {
360          this.value = value;
361        }
362    
363        @JsonValue
364        public String getValue() {
365          return this.value;
366        }
367    
368        @JsonCreator
369        public static Type create( final String value ) {
370          for( Type type : Type.values() ) {
371            if( type.getValue().equals( value ) ) {
372              return type;
373            }
374          }
375          throw new IllegalArgumentException( "Invalid value for Payment.CardType" );
376        }
377      }

407      public final static class Filter {
408    
409        @SnakeCase( "card_type" )
410        private String cardType;
411    
412        @SnakeCase( "created_at" )
413        private String createdAt;
414    
415        private Filter() {
416          super();
417        }
418    
419        public Payment.Filter byCardType( final Payment.CardType cardType ) {
420          this.cardType = cardType.getValue();
421          return this;
422        }
423    
424        public Payment.Filter byCreatedAt( final Date startCreatedAt, final Date endCreatedAt ) {
425          this.createdAt = String.valueOf( startCreatedAt.getTime() ) + "-" + String.valueOf( endCreatedAt.getTime() );
426          return this;
427        }
428      }
