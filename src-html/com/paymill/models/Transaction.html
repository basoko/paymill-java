274      public final static class Filter {
275    
276        @SnakeCase( "client" )
277        private String clientId;
278    
279        @SnakeCase( "payment" )
280        private String paymentId;
281    
282        @SnakeCase( "amount" )
283        private String amount;
284    
285        @SnakeCase( "description" )
286        private String description;
287    
288        @SnakeCase( "created_at" )
289        private String createdAt;
290    
291        @SnakeCase( "updated_at" )
292        private String updatedAt;
293    
294        @SnakeCase( "status" )
295        private String status;
296    
297        private Filter() {
298          super();
299        }
300    
301        public Transaction.Filter byClientId( final String clientId ) {
302          this.clientId = clientId;
303          return this;
304        }
305    
306        public Transaction.Filter byPaymentId( final String paymentId ) {
307          this.paymentId = paymentId;
308          return this;
309        }
310    
311        public Transaction.Filter byAmount( final int amount ) {
312          this.amount = String.valueOf( amount );
313          return this;
314        }
315    
316        public Transaction.Filter byAmountGreaterThan( final int amount ) {
317          this.amount = ">" + String.valueOf( amount );
318          return this;
319        }
320    
321        public Transaction.Filter byAmountLessThan( final int amount ) {
322          this.amount = "<" + String.valueOf( amount );
323          return this;
324        }
325    
326        public Transaction.Filter byDescription( final String description ) {
327          this.description = description;
328          return this;
329        }
330    
331        public Transaction.Filter byCreatedAt( final Date startCreatedAt, final Date endCreatedAt ) {
332          this.createdAt = String.valueOf( startCreatedAt.getTime() ) + "-" + String.valueOf( endCreatedAt.getTime() );
333          return this;
334        }
335    
336        public Transaction.Filter byUpdatedAt( final Date startCreatedAt, final Date endCreatedAt ) {
337          this.updatedAt = String.valueOf( startCreatedAt.getTime() ) + "-" + String.valueOf( endCreatedAt.getTime() );
338          return this;
339        }
340    
341        public Transaction.Filter byStatus( final Transaction.Status status ) {
342          this.status = status.getValue();
343          return this;
344        }

381      public enum Status {
382    
383        OPEN("open"),
384    
385        PENDING("pending"),
386    
387        CLOSED("closed"),
388    
389        FAILED("failed"),
390    
391        PARTIAL_REFUNDED("partial_refunded"),
392    
393        REFUNDED("refunded"),
394    
395        PREAUTH("preauth"),
396    
397        CHARGEBACK("chargeback");
398    
399        private String value;
400    
401        private Status( final String value ) {
402          this.value = value;
403        }
404    
405        @JsonValue
406        public String getValue() {
407          return this.value;
408        }
409    
410        @JsonCreator
411        public static Status create( final String value ) {
412          for( Status status : Status.values() ) {
413            if( status.getValue().equals( value ) ) {
414              return status;
415            }
416          }
417          throw new IllegalArgumentException( "Invalid value for Transaction.Status" );
418        }
419      }
