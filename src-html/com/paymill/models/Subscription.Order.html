009    /**
010     * Subscriptions allow you to charge recurring payments on a client’s credit card / to a client’s direct debit. A subscription
011     * connects a client to the {@link Offer}s. A client can have several subscriptions to different offers, but only one subscription
012     * to the same offer.
013     * @author Vassil Nikolov
014     * @since 3.0.0
015     */

302      public final static class Filter {
303    
304        @SnakeCase( "offer" )
305        private String offerId;
306    
307        @SnakeCase( "created_at" )
308        private String createdAt;
309    
310        private Filter() {
311          super();
312        }
313    
314        public Subscription.Filter byOfferId( final String offerId ) {
315          this.offerId = offerId;
316          return this;
317        }
318    
319        public Subscription.Filter byCreatedAt( final Date startCreatedAt, final Date endCreatedAt ) {
320          this.createdAt = String.valueOf( startCreatedAt.getTime() ) + "-" + String.valueOf( endCreatedAt.getTime() );
321          return this;
322        }
323    
324      }
325    
326      public final static class Order {
327    
328        @SnakeCase( "offer" )
329        private boolean offer;
330    
331        @SnakeCase( "canceled_at" )
332        private boolean canceledAt;
333    
334        @SnakeCase( "created_at" )
335        private boolean createdAt;
336    
337        @SnakeCase( value = "asc", order = true )
338        private boolean asc;
339    
340        @SnakeCase( value = "desc", order = true )
341        private boolean desc;
342    
343        private Order() {
344          super();
345        }
346    
347        public Subscription.Order asc() {
348          this.asc = true;
349          this.desc = false;
350          return this;
351        }
352    
353        public Subscription.Order desc() {
354          this.asc = false;
355          this.desc = true;
356          return this;
357        }
358    
359        public Subscription.Order byOffer() {
360          this.offer = true;
361          this.createdAt = false;
362          this.canceledAt = false;
363          return this;
364        }
365    
366        public Subscription.Order byCanceledAt() {
367          this.offer = false;
368          this.createdAt = false;
369          this.canceledAt = true;
370          return this;
371        }
372    
373        public Subscription.Order byCreatedAt() {
374          this.offer = false;
375          this.createdAt = true;
376          this.canceledAt = false;
377          return this;
378        }
379    
380      }
