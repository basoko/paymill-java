307      public enum Status {
308        OPEN("open"), REFUNDED("refunded"), FAILED("failed");
309    
310        private String value;
311    
312        private Status( final String value ) {
313          this.value = value;
314        }
315    
316        @JsonValue
317        public String getValue() {
318          return this.value;
319        }
320    
321        @JsonCreator
322        public static Status create( final String value ) {
323          for( Status status : Status.values() ) {
324            if( status.getValue().equals( value ) ) {
325              return status;
326            }
327          }
328          throw new IllegalArgumentException( "Invalid value for Refund.Status" );
329        }
