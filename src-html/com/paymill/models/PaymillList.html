001    package com.paymill.models;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
007    import com.fasterxml.jackson.annotation.JsonProperty;
008    
009    @JsonIgnoreProperties( ignoreUnknown = true )
010    public final class PaymillList<T> {
011    
012      public PaymillList() {
013        this.data = new ArrayList<T>();
014      }
015    
016      private List<T> data;
017    
018      @JsonProperty( "data_count" )
019      private int     dataCount;
020    
021      public List<T> getData() {
022        return this.data;
023      }
024    
025      public void setData( final List<T> data ) {
026        this.data = data;
027      }
028    
029      public int getDataCount() {
030        return this.dataCount;
031      }
032    
033      public void setDataCount( final int dataCount ) {
034        this.dataCount = dataCount;
035      }
036    
037    }
