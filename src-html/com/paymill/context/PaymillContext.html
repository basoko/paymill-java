001    package com.paymill.context;
002    
003    import java.lang.reflect.Constructor;
004    import java.util.Date;
005    
006    import org.apache.commons.beanutils.ConvertUtils;
007    import org.apache.commons.beanutils.converters.DateConverter;
008    import org.apache.commons.lang3.StringUtils;
009    
010    import com.fasterxml.jackson.databind.ObjectMapper;
011    import com.paymill.services.ClientService;
012    import com.paymill.services.OfferService;
013    import com.paymill.services.PaymentService;
014    import com.paymill.services.PreauthorizationService;
015    import com.paymill.services.RefundService;
016    import com.paymill.services.SubscriptionService;
017    import com.paymill.services.TransactionService;
018    import com.paymill.services.WebhookService;
019    import com.sun.jersey.api.client.Client;
020    import com.sun.jersey.api.client.filter.HTTPBasicAuthFilter;
021    
022    /**
023     * PaymillContecxt loads the context of PAYMILL for a single account, by providing a merchants private key<br />
024     * It creates 8 services, which represents the PAYMILL API:
025     * <ul>
026     * <li>{@link ClientService}</li>
027     * <li>{@link OfferService}</li>
028     * <li>{@link PaymentService}</li>
029     * <li>{@link PreauthorizationService}</li>
030     * <li>{@link RefundService}</li>
031     * <li>{@link SubscriptionService}</li>
032     * <li>{@link TransactionService}</li>
033     * <li>{@link WebhookService}</li>
034     * </ul>
035     * This services should not be created directly. They have to be obtained by the context's accessors.
036     * @author Vassil Nikolov
037     * @since 3.0.0
038     */
039    public final class PaymillContext {
040    
041      private ClientService           clientService;
042      private OfferService            offerService;
043      private PaymentService          paymentService;
044      private PreauthorizationService preauthorizationService;
045      private RefundService           refundService;
046      private SubscriptionService     subscriptionService;
047      private TransactionService      transactionService;
048      private WebhookService          webhookService;
049    
050      private Client                  httpClient;
051    
052      private static ObjectMapper     jacksonParser = new ObjectMapper();
053    
054      /**
055       * Creates a PAYMILL context with the given apiKey. Connection timeout to PAYMILL by default is set to infinity.
056       * @param apiKey
057       *          Private key from PAYMILL merchant center.
058       */
059      public PaymillContext( final String apiKey ) {
060        this( apiKey, null );
061      }
062    
063      /**
064       * Creates a PAYMILL context with the given apiKey and timeout.
065       * @param apiKey
066       *          Private key from PAYMILL merchant center.
067       * @param timeout
068       *          Timeout in milliseconds for the HTTP connection to PAYMILL. If <code>null</code> or <code>0</code> then an interval
069       *          of infinity is declared.
070       */
071      public PaymillContext( final String apiKey, Integer timeout ) {
072        ConvertUtils.register( new DateConverter( null ), Date.class );
073        try {
074          this.httpClient = new Client();
075          this.httpClient.setReadTimeout( timeout );
076          this.httpClient.setConnectTimeout( timeout );
077          this.httpClient.addFilter( new HTTPBasicAuthFilter( apiKey, StringUtils.EMPTY ) );
078    
079          this.clientService = this.getPrivateConstructor( ClientService.class ).newInstance( this.httpClient );
080          this.offerService = this.getPrivateConstructor( OfferService.class ).newInstance( this.httpClient );
081          this.paymentService = this.getPrivateConstructor( PaymentService.class ).newInstance( this.httpClient );
082          this.preauthorizationService = this.getPrivateConstructor( PreauthorizationService.class ).newInstance( this.httpClient );
083          this.refundService = this.getPrivateConstructor( RefundService.class ).newInstance( this.httpClient );
084          this.subscriptionService = this.getPrivateConstructor( SubscriptionService.class ).newInstance( this.httpClient );
085          this.transactionService = this.getPrivateConstructor( TransactionService.class ).newInstance( this.httpClient );
086          this.webhookService = this.getPrivateConstructor( WebhookService.class ).newInstance( this.httpClient );
087        } catch( Exception exc ) {
088          throw new RuntimeException( exc );
089        }
090      }
091    
092      public static ObjectMapper getJacksonParser() {
093        return PaymillContext.jacksonParser;
094      }
095    
096      public ClientService getClientService() {
097        return this.clientService;
098      }
099    
100      public OfferService getOfferService() {
101        return this.offerService;
102      }
103    
104      public PaymentService getPaymentService() {
105        return this.paymentService;
106      }
107    
108      public PreauthorizationService getPreauthorizationService() {
109        return this.preauthorizationService;
110      }
111    
112      public RefundService getRefundService() {
113        return this.refundService;
114      }
115    
116      public SubscriptionService getSubscriptionService() {
117        return this.subscriptionService;
118      }
119    
120      public TransactionService getTransactionService() {
121        return this.transactionService;
122      }
123    
124      public WebhookService getWebhookService() {
125        return this.webhookService;
126      }
127    
128      private <T> Constructor<T> getPrivateConstructor( final Class<T> clazz ) throws Exception {
129        Constructor<T> declaredConstructor = clazz.getDeclaredConstructor( Client.class );
130        declaredConstructor.setAccessible( true );
131        return declaredConstructor;
132      }
133    
134    }
